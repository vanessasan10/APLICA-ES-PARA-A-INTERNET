
CREATE DATABASE sistema_tarefas;

USE sistema_tarefas;

CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    email VARCHAR(100) UNIQUE,
    senha VARCHAR(255)
);



CREATE TABLE tarefas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(100),
    descricao TEXT,
    status VARCHAR(20),
    usuario_id INT,
    FOREIGN KEY(usuario_id)
    REFERENCES usuarios(id)
);
